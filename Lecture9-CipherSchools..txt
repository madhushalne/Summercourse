//inheritance

import java.util.*;

class Shape{
double pi= 3.14;
double getSquare(int r)
{
return r*r;
}
}
class circle extends Shape 
{
void area(int r)
{
System.out.println(pi*getSquare(r));
}
}
public class InheritanceStudy{
public static void main(String[] args)
{
circle c = new circle();
circle.area(5);
}
}
